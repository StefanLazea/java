package ro.ase.java.models;

public class Vehicle {
	// modificatorul default - vizibil la nivel de pachet
	private String make;
	private String model;
	private int cubicCapacity;
	private float weight;
	private double price;
	private boolean electricVehicle;
	
	//Metode de acces
	
	public Vehicle()   //constructor default
	{
		
	}
	
	public Vehicle(String make, 
					String model,
					int cc,
					float weight,
					double price,
					boolean isElectric)
	{	
		this.make = make;
		this.model = model;
		this.cubicCapacity = cc;
		this.weight = weight;
		this.price = price;
		this.electricVehicle = isElectric;	
	}
	
	public void setMake(String make)
	{
		this.make=make;
	}
	public void setModel(String model)
	{
		this.model=model;
	}
	public void setCubicCapacity( int cc)
	{
		this.cubicCapacity=cc;
	}
	public void setWeight(float weight)
	{
		this.weight=weight;
	}
	public void setPrice(double price)
	{
		this.price=price;
	}
	public void setElectricVehicle(boolean isElectric)
	{
		this.electricVehicle=isElectric;
	}
	
	public String getMake()
	{
		return this.make;
	}
	public String getModel()
	{
		return this.model;
	}
	
	public int getCubicCapacity()
	{
		return this.cubicCapacity;
	}
	public float getWeight()
	{
		return this.weight;
	}
	public double getPrice()
	{
		return this.price;
	}
	public boolean getIsElectric()
	{
		return this.electricVehicle;
	}
	
}
